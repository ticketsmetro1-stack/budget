// Basic offline cache for index + assets const CACHE_NAME = 'budget-v1'; const FILES = [ './', './index.html', './icon.png' ]; self.addEventListener('install', evt => { evt.waitUntil( caches.open(CACHE_NAME).then(cache => cache.addAll(FILES)) ); }); self.addEventListener('activate', evt => { evt.waitUntil( caches.keys().then(keys => Promise.all( keys.map(k => { if (k !== CACHE_NAME) return caches.delete(k); }) )) ); }); self.addEventListener('fetch', evt => { evt.respondWith( caches.match(evt.request).then(resp => resp || fetch(evt.request)) ); });
